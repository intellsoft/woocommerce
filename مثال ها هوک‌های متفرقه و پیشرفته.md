### ۱. `woocommerce_init` — بارگذاری اولیهٔ ووکامرس (Action)

```php
/**
 * راه‌اندازی کلاس‌های شخص ثالث پس از آماده شدن هستهٔ ووکامرس
 */
add_action( 'woocommerce_init', 'my_plugin_initialize' );
function my_plugin_initialize() {
	// مثلاً نمونه‌سازی یک کلاس مدیریت موجودی
	if ( class_exists( 'My_Inventory_Manager' ) ) {
		new My_Inventory_Manager();
	}
}
```

### ۳. `woocommerce_product_query` — تغییر کوئری حلقهٔ محصولات (Action)

```php
/**
 * فیلتر محصولات بر اساس متای سفارشی (مثلاً فقط محصولات ویژه)
 */
add_action( 'woocommerce_product_query', 'filter_products_by_featured' );
function filter_products_by_featured( $q ) {
	if ( is_shop() || is_product_category() ) {
		$meta_query = $q->get( 'meta_query' );
		$meta_query[] = array(
			'key'     => '_featured',
			'value'   => 'yes',
			'compare' => '='
		);
		$q->set( 'meta_query', $meta_query );
	}
}
```

### ۷. `woocommerce_before_template_part` — قبل از قالب جزئی (Action)

```php
/**
 * اگر قالب "content-single-product" بارگذاری می‌شود، یک پرچم تنظیم کن
 */
add_action( 'woocommerce_before_template_part', 'mark_single_product_template', 10, 4 );
function mark_single_product_template( $template_name, $template_path, $located, $args ) {
	if ( 'content-single-product.php' === $template_name ) {
		// یک متغیر سراسری برای استفاده در بخش‌های دیگر
		$GLOBALS['is_single_product_template'] = true;
	}
}
```

### ۱۵. `woocommerce_cart_calculate_fees` — افزودن هزینه اضافی (Action)

```php
/**
 * اضافه کردن هزینهٔ بسته‌بندی 5 هزار تومان برای سفارش‌های زیر 200 هزار
 */
add_action( 'woocommerce_cart_calculate_fees', 'add_packaging_fee' );
function add_packaging_fee() {
	if ( is_admin() && ! defined( 'DOING_AJAX' ) ) return;
	$cart_total = WC()->cart->cart_contents_total;
	if ( $cart_total < 200000 ) {
		WC()->cart->add_fee( 'هزینه بسته‌بندی', 5000 );
	}
}
```

### ۲۲. `woocommerce_rest_insert_product` — ایجاد/به‌روزرسانی محصول از REST API (Action)

```php
/**
 * ثبت یک متا هنگام ایجاد محصول از طریق API (مثلاً منبع ورود)
 */
add_action( 'woocommerce_rest_insert_product', 'track_api_product_creation', 10, 2 );
function track_api_product_creation( $product, $request ) {
	if ( 'POST' === $request->get_method() ) {
		$product->update_meta_data( '_imported_via_api', current_time( 'mysql' ) );
		$product->save_meta_data();
	}
}
```
