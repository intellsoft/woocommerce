در این بخش، برای هر یک از هوک‌های مربوط به حساب کاربری و فرم‌های ورود/ثبت‌نام، یک یا چند مثال کاملاً عملی همراه با توضیح فارسی ارائه می‌شود.

## ۱. افزودن پیام خوش‌آمدگویی در بالای منوی حساب کاربری  
### هوک: `woocommerce_before_account_navigation`

```php
/**
 * نمایش یک پیام خوش‌آمدگویی شخصی‌سازی‌شده قبل از منوی ناوبری حساب
 */
add_action( 'woocommerce_before_account_navigation', 'add_welcome_message_before_nav' );
function add_welcome_message_before_nav() {
	$current_user = wp_get_current_user();
	$display_name = $current_user->display_name;
	echo '<div class="account-welcome" style="padding: 10px; background: #f0f0f0; margin-bottom: 15px;">';
	echo '👋 ' . sprintf( 'سلام %s، به حساب کاربری خود خوش آمدید!', $display_name );
	echo '</div>';
}
```

## ۳. شخصی‌سازی کامل داشبورد حساب کاربری  
### هوک: `woocommerce_account_dashboard`

```php
/**
 * جایگزینی داشبورد پیش‌فرض با محتوای دلخواه
 */
add_action( 'woocommerce_account_dashboard', 'custom_dashboard_content' );
function custom_dashboard_content() {
	$current_user = wp_get_current_user();
	echo '<p>از اینجا می‌توانید سفارش‌های اخیر، اطلاعات حساب و جزئیات عضویت خود را مشاهده کنید.</p>';
	// نمایش لینک‌های سریع
	echo '<div class="quick-links">';
	echo '<a href="' . wc_get_account_endpoint_url( 'orders' ) . '" class="button">مشاهده سفارش‌ها</a> ';
	echo '<a href="' . wc_get_account_endpoint_url( 'edit-account' ) . '" class="button">ویرایش اطلاعات</a>';
	echo '</div>';
}
```

> اگر فقط می‌خواهید **چیزی به داشبورد اضافه کنید** (نه جایگزینی کامل)، می‌توانید از اولویت بالاتر استفاده کنید یا از همان هوک بدون حذف محتوای پیش‌فرض (که در اولویت 10 قرار دارد) با اولویت 11 محتوا اضافه کنید.

## ۵. اضافه کردن یک دکمهٔ «درخواست مرجوعی» بعد از جدول سفارشات  
### هوک: `woocommerce_after_account_orders`

```php
/**
 * بعد از جدول سفارشات یک دکمه برای فرم مرجوعی قرار می‌دهد
 */
add_action( 'woocommerce_after_account_orders', 'add_return_request_button' );
function add_return_request_button() {
	if ( has_orders() ) {
		echo '<a href="' . site_url( '/request-return' ) . '" class="button alt">درخواست مرجوعی کالا</a>';
	}
}
function has_orders() {
	$customer_orders = wc_get_orders( array(
		'customer' => get_current_user_id(),
		'limit'    => 1,
	) );
	return ! empty( $customer_orders );
}
```

## ۸. اضافه کردن فیلد «شماره موبایل» در فرم ثبت‌نام  
### هوک‌های: `woocommerce_register_form_start` / `woocommerce_register_form_end`  
### هوک پردازش: `woocommerce_created_customer`

```php
/**
 * ۱. نمایش فیلد موبایل در فرم ثبت‌نام
 */
add_action( 'woocommerce_register_form', 'add_phone_field_to_registration' );
function add_phone_field_to_registration() {
	?>
	<p class="form-row form-row-wide">
		<label for="reg_phone">شماره موبایل <span class="required">*</span></label>
		<input type="text" class="input-text" name="phone" id="reg_phone" value="<?php if ( ! empty( $_POST['phone'] ) ) echo esc_attr( $_POST['phone'] ); ?>" />
	</p>
	<?php
}

/**
 * ۲. اعتبارسنجی فیلد
 */
add_filter( 'woocommerce_registration_errors', 'validate_phone_field', 10, 3 );
function validate_phone_field( $errors, $username, $email ) {
	if ( isset( $_POST['phone'] ) && empty( $_POST['phone'] ) ) {
		$errors->add( 'phone_error', 'لطفاً شماره موبایل خود را وارد نمایید.' );
	} elseif ( ! preg_match( '/^09[0-9]{9}$/', $_POST['phone'] ) ) {
		$errors->add( 'phone_error', 'شماره موبایل معتبر نیست.' );
	}
	return $errors;
}

/**
 * ۳. ذخیرهٔ شماره موبایل در پروفایل کاربر پس از ثبت‌نام
 */
add_action( 'woocommerce_created_customer', 'save_phone_number_after_registration' );
function save_phone_number_after_registration( $customer_id ) {
	if ( isset( $_POST['phone'] ) ) {
		update_user_meta( $customer_id, 'phone', sanitize_text_field( $_POST['phone'] ) );
		// می‌توانید آن را به billing phone هم متصل کنید
		update_user_meta( $customer_id, 'billing_phone', sanitize_text_field( $_POST['phone'] ) );
	}
}
```

> **توجه:** `woocommerce_register_form_start` و `_end` معمولاً برای افزودن محتوای ایستا استفاده می‌شوند؛ برای افزودن فیلدها بهتر است از خود `woocommerce_register_form` (که بین start و end قرار دارد) استفاده شود. مثال بالا این روش را نشان می‌دهد.

## ۹. سفارشی‌سازی منوی حساب کاربری: تغییر ترتیب، نام و افزودن آیتم جدید  
### هوک: `woocommerce_account_menu_items`

```php
/**
 * افزودن یک آیتم «تیکت‌های پشتیبانی» به منوی حساب
 */
add_filter( 'woocommerce_account_menu_items', 'customize_account_menu', 10, 1 );
function customize_account_menu( $items ) {
	// حذف آیتم دانلودها (اگر نیاز ندارید)
	unset( $items['downloads'] );

	// تغییر نام یک آیتم
	$items['orders'] = 'سفارش‌های من';

	// افزودن آیتم جدید (توجه: باید endpoint مربوطه را هم ثبت کنید)
	$items['support-tickets'] = 'تیکت‌های پشتیبانی';

	return $items;
}

// ثبت endpoint جدید برای آیتم منو (مهم)
add_action( 'init', 'add_support_tickets_endpoint' );
function add_support_tickets_endpoint() {
	add_rewrite_endpoint( 'support-tickets', EP_ROOT | EP_PAGES );
}

// محتوای صفحهٔ endpoint
add_action( 'woocommerce_account_support-tickets_endpoint', 'support_tickets_content' );
function support_tickets_content() {
	echo '<p>در اینجا لیست تیکت‌های شما قرار می‌گیرد.</p>';
}

// بعد از افزودن کد، یکبار پیوندهای یکتا (permalinks) را ذخیره کنید.
```

## ۱۰. حذف ستون «قیمت کل» از جدول سفارشات حساب کاربری  
### هوک: `woocommerce_my_account_my_orders_columns`

```php
/**
 * حذف ستون «قیمت کل» و اضافه کردن ستون «روش پرداخت»
 */
add_filter( 'woocommerce_my_account_my_orders_columns', 'customize_orders_columns' );
function customize_orders_columns( $columns ) {
	// حذف قیمت کل
	unset( $columns['order-total'] );
	// افزودن ستون روش پرداخت
	$columns['payment-method'] = 'روش پرداخت';
	return $columns;
}

// برای پر کردن ستون جدید
add_action( 'woocommerce_my_account_my_orders_column_payment-method', 'fill_payment_method_column' );
function fill_payment_method_column( $order ) {
	echo esc_html( $order->get_payment_method_title() );
}
```

## ۱۱. تغییر مسیر کاربر پس از ورود موفق  
### هوک: `woocommerce_login_redirect`

```php
/**
 * هدایت مشتری به صفحهٔ «محصولات ویژه» پس از ورود
 */
add_filter( 'woocommerce_login_redirect', 'redirect_after_login', 10, 2 );
function redirect_after_login( $redirect, $user ) {
	// اگر کاربر نقش مدیر یا فروشنده نباشد
	if ( in_array( 'customer', $user->roles ) || in_array( 'subscriber', $user->roles ) ) {
		return site_url( '/special-offers' ); // آدرس صفحهٔ دلخواه
	}
	return $redirect; // برای دیگر نقش‌ها مسیر پیش‌فرض
}
```

## ۱۲. افزودن یک اکشن سفارشی در صفحهٔ محتوای حساب (برای endpointهای دلخواه)  
### هوک: `woocommerce_account_content`

اگر endpoint اختصاصی ندارید اما می‌خواهید در تمام صفحات حساب محتوایی مشترک اضافه کنید:

```php
/**
 * افزودن یک نوار اطلاع‌رسانی بالای محتوای حساب کاربری
 */
add_action( 'woocommerce_account_content', 'add_account_top_notice', 5 );
function add_account_top_notice() {
	if ( is_wc_endpoint_url( 'orders' ) ) {
		echo '<div class="woocommerce-info">سفارش‌های شما در این بخش قابل پیگیری است.</div>';
	}
}
```
