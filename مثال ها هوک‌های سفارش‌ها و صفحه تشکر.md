## ۶. نمایش یادداشت سفارش در صفحهٔ جزئیات سفارش (حساب کاربری)  
### هوک: `woocommerce_view_order`

```php
/**
 * هنگام مشاهدهٔ یک سفارش در حساب کاربری، یادداشت‌های خصوصی مدیر را به مشتری نشان می‌دهد
 * (در صورتی که مجاز باشد)
 */
add_action( 'woocommerce_view_order', 'show_order_notes_to_customer', 5, 1 );
function show_order_notes_to_customer( $order_id ) {
	$order = wc_get_order( $order_id );
	$notes = wc_get_order_notes( array( 'order_id' => $order_id ) );
	if ( ! empty( $notes ) ) {
		echo '<h3>یادداشت‌های سفارش</h3><ul>';
		foreach ( $notes as $note ) {
			// فقط یادداشت‌هایی که خطاب به مشتری هستند (customer_note = true)
			if ( $note->customer_note ) {
				echo '<li>' . wpautop( wptexturize( $note->content ) ) . '</li>';
			}
		}
		echo '</ul>';
	}
}
```

## ۱۳. جلوگیری از ذخیرهٔ سفارش در صورت عدم موجودی کافی  
### هوک: `woocommerce_before_order_object_save`

```php
/**
 * قبل از ذخیرهٔ سفارش، بررسی می‌کند که همهٔ محصولات موجود باشند
 * و در غیر این صورت یک خطا صادر می‌کند
 * @param WC_Order $order
 */
add_action( 'woocommerce_before_order_object_save', 'check_stock_before_save', 10, 1 );
function check_stock_before_save( $order ) {
	foreach ( $order->get_items() as $item ) {
		$product = $item->get_product();
		if ( $product && ! $product->is_in_stock() ) {
			throw new Exception( sprintf( 'محصول "%s" در انبار موجود نیست.', $product->get_name() ) );
		}
	}
}
```

## ۱۴. تغییر شماره سفارش نمایش داده‌شده (افزودن پیشوند)

/**
 * به جای شمارهٔ پیش‌فرض (ID)، یک شمارهٔ سفارش سفارشی با پیشوند تاریخ ایجاد می‌کند

	return 'WC-' . date( 'Ymd' ) . '-' . $order->get_id();
}

## ۱۶. افزودن دکمهٔ «درخواست مرجوعی» به لیست سفارش‌های حساب کاربری  
### هوک: `woocommerce_my_account_my_orders_actions`

```php
/**
 * یک دکمه جدید به هر سفارش در حساب کاربری اضافه می‌کند
 * @param array    $actions دکمه‌های موجود
 * @param WC_Order $order
 */
add_filter( 'woocommerce_my_account_my_orders_actions', 'add_return_request_button', 10, 2 );
function add_return_request_button( $actions, $order ) {
	// فقط برای سفارش‌های تکمیل شده
	if ( $order->has_status( 'completed' ) ) {
		$actions['return'] = array(
			'url'  => wp_nonce_url( add_query_arg( 'request_return', $order->get_id() ), 'return_request' ),
			'name' => 'درخواست مرجوعی',
		);
	}
	return $actions;
}
```
